    function switchTab(tabName) {
      const tabs = document.querySelectorAll('.tab-content');
      const buttons = document.querySelectorAll('.tab-btn');
      
      tabs.forEach(tab => {
        tab.classList.remove('active');
      });
      
      buttons.forEach(btn => {
        btn.classList.remove('active');
      });
      
      document.getElementById(tabName).classList.add('active');
      
      // ä½¿ç”¨ data-target ä¾†æŸ¥æ‰¾æŒ‰éˆ•ï¼Œæ›´ç©©å®š
      const activeButton = document.querySelector(`.tab-btn[data-target="${tabName}"]`);
      if (activeButton) {
        activeButton.classList.add('active');
      } else {
        // Fallback for footer button or if data-target is missing
        const fallbackBtn = Array.from(buttons).find(btn => 
           btn.getAttribute('onclick').includes(tabName) && !btn.classList.contains('active')
        );
        if(fallbackBtn) fallbackBtn.classList.add('active');
      }
    }
